## 先去学习webpackgithub官网，把webpack下载下来
### 然后在下载的目录创建目录hy-mjlcode
### 配置webpack.config.js
```
const path = require('path')

module.exports = {
  mode: 'development',
  context: __dirname,
  entry: './src/index.js',
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: 'bundle.js',
    clean: true,
  },
}

```

### 在hy-mjlcode目录下创建src目录和index.js文件
```
index.js
const webpack = require('../lib/webpack')
const config = require('./webpack.config')

const compiler = webpack(config)

compiler.run((err, stats) => {
  if (err) {
    console.error(err)
    return
  }

  if (stats?.hasErrors()) {
    console.error(stats.toString({ colors: true }))
    compiler.close(() => {})
    return
  }

  console.log(stats.toString({ colors: true }))
  compiler.close(() => {})
})

```
### 在src/index.js文件中写入一些代码
```
console.log('hy-mjlcode: webpack run test')

```


### 在hy-mjlcode目录下执行npm run build
